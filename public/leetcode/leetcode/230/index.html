230. 二叉搜索树中第K小的元素

原题链接

深度优先遍历，遍历的时候携带要找第几小即可：

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int kthSmallest(TreeNode* root, int k) {
        int child_cnt;
        return kthSmallest(root, k, child_cnt);
    }

private:
    int kthSmallest(TreeNode* root, int k, int& child_cnt) {
        if (root == nullptr) {
            child_cnt = 0;
            return -1;
        }

        int l_child_cnt = 0, r_child_cnt = 0;
        int l_ret = kthSmallest(root->left, k, l_child_cnt);
        if (l_ret != -1) {
            return l_ret;
        }
        if (l_child_cnt+1 == k) {
            return root->val;
        }
        int r_ret = kthSmallest(root->right, k-l_child_cnt-1, r_child_cnt);
        if (r_ret != -1) {
            return r_ret;
        }
        child_cnt = l_child_cnt + r_child_cnt + 1;
        return -1;
    }
};

因为是有序的搜索二叉树，可以用扫描出中序遍历即可，中序遍历中的第k个元素即是第k大的元素：

class Solution {
public:
    int kthSmallest(TreeNode* root, int k) {
        stack<TreeNode*> st;
        while (!st.empty() || root) {
            while (root) {
                st.push(root);
                root = root->left;
            }
            root = st.top();
            st.pop();

            if (--k == 0) {
                return root->val;
            }
            root = root->right;
        }
        return 0;
    }
};
