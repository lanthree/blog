138. 复制带随机指针的链表

原题链接

/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* next;
    Node* random;
    
    Node(int _val) {
        val = _val;
        next = NULL;
        random = NULL;
    }
};
*/

class Solution {
public:
    Node* copyRandomList(Node* head) {
        if (head == nullptr) {
            return nullptr;
        }

        Node dummy_old(0);
        dummy_old.next = head;

        Node dummy_new(0);
        Node* tail_new = &dummy_new;

        // 根据next新建链表 并记录结点对应关系
        unordered_map<unsigned long long, unsigned long long> node_mapping;
        for (Node* curr = dummy_old.next; curr; curr = curr->next) {
            tail_new->next = new Node(curr->val);
            node_mapping[(unsigned long long)curr] = (unsigned long long)tail_new->next;
            tail_new = tail_new->next;
        }

        // 根据结点对应关系，恢复random结构
        for (Node* curr_new = dummy_new.next, * curr_old = dummy_old.next; 
             curr_new;
             curr_new = curr_new->next, curr_old = curr_old->next) {
            if (curr_old->random) {
                curr_new->random = (Node*)node_mapping[(unsigned long long)curr_old->random];
            }
        }

        return dummy_new.next;
    }
};

首先，上述解法中unordered_map的模版参数可更换为Node*，就不会有很多C峰哥的强制转换。

另外，上述解法中，在第一次迭代构建链表时，因random可能指向还为构建的位置，所以用了两次遍历；有一个跟两次遍历等效的方法，是 递归构建链表 回溯更新randowm。

class Solution {
public:
    unordered_map<Node*, Node*> cachedNode;

    Node* copyRandomList(Node* head) {
        if (head == nullptr) {
            return nullptr;
        }
        if (!cachedNode.count(head)) {
            Node* headNew = new Node(head->val);
            cachedNode[head] = headNew;
            headNew->next = copyRandomList(head->next);
            headNew->random = copyRandomList(head->random);
        }
        return cachedNode[head];
    }
};

为了节省，hash表的空间，我们可以先把链表的每个节点的新副本挂在原节点后面，然后修改新结点的random指针、然后修改新结点的next指针：

class Solution {
public:
    Node* copyRandomList(Node* head) {
        if (head == nullptr) {
            return nullptr;
        }
        for (Node* node = head; node != nullptr; node = node->next->next) {
            Node* nodeNew = new Node(node->val);
            nodeNew->next = node->next;
            node->next = nodeNew;
        }
        for (Node* node = head; node != nullptr; node = node->next->next) {
            Node* nodeNew = node->next;
            nodeNew->random = (node->random != nullptr) ? node->random->next : nullptr;
        }
        Node* headNew = head->next;
        for (Node* node = head; node != nullptr; node = node->next) {
            Node* nodeNew = node->next;
            node->next = node->next->next;
            nodeNew->next = (nodeNew->next != nullptr) ? nodeNew->next->next : nullptr;
        }
        return headNew;
    }
};
