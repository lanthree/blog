300. 最长递增子序列

原题链接

令dp[i]表示以数字nums[i]结尾的串的最长递增子序列，容易写出O(n^2)的代码：

class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int dp[2500];
        for (int i = 0; i < nums.size(); i++) {
            dp[i] = 1;
        }

        int ans = 1;
        for (int j = 1; j < nums.size(); j++) {
            for (int i = 0; i < j; i++) {
                if (nums[j] > nums[i]) {
                    dp[j] = max(dp[j], dp[i]+1);
                    ans = max(ans, dp[j]);
                }
            }
        }
        return ans;
    }
};

官方给出了一个O(nlogn)的方法：贪心 + 二分查找。

class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int len = 1, n = (int)nums.size();
        if (n == 0) {
            return 0;
        }
        // d[i] 表示程度为i的字串 末尾数字的最小值
        vector<int> d(n + 1, 0);
        d[len] = nums[0];
        // 遍历数字
        for (int i = 1; i < n; ++i) {
            // 如果当前数字 比 当前最长长度的末尾数字大
            if (nums[i] > d[len]) {
                // 则长度+1的末尾数字是该数字
                d[++len] = nums[i];

            } else {
                // 如果找不到说明所有的数都比 nums[i] 大，此时要更新 d[1]，所以这里将 pos 设为 0
                int pos = 0;
                int l = 1, r = len;
                while (l <= r) {
                    int mid = (l + r) >> 1;
                    if (d[mid] < nums[i]) {
                        // 找 pos < nums[i] < pos+1 的位置
                        // 然后用nums[i]更新d[pos+1]
                        pos = mid;
                        l = mid + 1;
                    } else {
                        r = mid - 1;
                    }
                }
                d[pos + 1] = nums[i];
            }
        }
        return len;
    }
};
