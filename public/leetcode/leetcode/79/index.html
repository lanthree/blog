class Solution {
    vector<vector<int>> dirs = {{1,0},{-1,0},{0,1},{0,-1}};
public:
    bool exist(vector<vector<char>>& board, string word) {
        int m = board.size();
        int n = board[0].size();

        vector<vector<bool>> vis(m, vector<bool>(n, false));
        for (int i = 0; i < m; ++i)
            for (int j = 0; j < n; ++j) {
                if (DFS(board, word, i, j, 0, vis)) {
                    return true;
                }
            }
        return false;
    }

private:
    bool DFS(vector<vector<char>>& board, string& word, int x, int y, int cur, vector<vector<bool>>& vis) {
        if (x >= board.size() || x < 0 || y >= board[0].size() || y < 0) {
            return false;
        } else if (vis[x][y]) {
            return false;
        } else if (board[x][y] != word[cur]) {
            return false;
        }

        if (cur+1 == word.length())
            return true;

        vis[x][y] = true;

        for (int i = 0; i < 4; ++i) {
            int n_x = x + dirs[i][0];
            int n_y = y + dirs[i][1];
            if (DFS(board, word, n_x, n_y, cur+1, vis)) {
                return true;
            }
        }
        vis[x][y] = false;
        return false;
    }
};
