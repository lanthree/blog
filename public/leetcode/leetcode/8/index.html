8. 字符串转换整数 (atoi)

原题链接

按描述编码的模拟题，貌似没有什么要注意的点。

class Solution {
public:
    int myAtoi(string s) {
        int ans = 0;

        int idx = 0;
        // 丢弃无用的前导空格
        while(idx < s.size() && s[idx] == ' ')
            idx++;
        
        if (idx >= s.size()) {
            return 0;
        }
        
        // 扫描正负符号
        bool is_neg = false;
        if (s[idx] == '-') {
            is_neg = true;
            idx++;
        } else if (s[idx] == '+') {
            idx++;
        }
        
        // 扫描接下来的数字
        while(idx < s.size() && isdigit(s[idx])) {
            int v = s[idx] - '0';
            v = is_neg ? -v : v;
            
            if (willOverFlow(ans, v)) {
                return is_neg ? INT_MIN : INT_MAX;
            }
            ans = ans*10+v;
            idx++;
        }
        
        return ans;
    }

private:
    bool willOverFlow(int ans, int addv) {
        if (ans < INT_MIN/10 || ans > INT_MAX/10) {
            return true;
        } else if (ans == INT_MIN/10 && addv < INT_MIN%10) {
            return true;
        } else if (ans == INT_MAX/10 && addv > INT_MAX%10) {
            return true;
        }
        return false;
    }
};

这里官方给了一个更高级的答案，有限状态自动机，如果状态管理比较复杂，用纯模拟的思路编码就会比较臃肿难以维护/阅读。这里官方给出的自动机解法对此题来说有点大材小用，但是是一个锻炼自动机理解的机会：

class Automaton {
    string state = "start";
    unordered_map<string, vector<string>> table = {
        {"start", {"start", "signed", "in_number", "end"}},
        {"signed", {"end", "end", "in_number", "end"}},
        {"in_number", {"end", "end", "in_number", "end"}},
        {"end", {"end", "end", "end", "end"}},
    };

    int get_col(char c) {
        if (c == ' ') return 0;
        if (c == '+' || c == '-') return 1;
        if (isdigit(c)) return 2;
        return 3;
    }

private:
    int sign = 1;
    long long val = 0;

public:

    void Add(char c) {
        state = table[state][get_col(c)];
        if (state == "in_number") {
            val = val*10 + (c-'0');
            val = sign == 1 ? min(val, (long long)INT_MAX) : min(val, -(long long)INT_MIN);
        } else if (state == "signed") {
            sign = c == '+' ? 1 : -1;
        }
    }

    int GetResult() {
        return (int)(val*sign);
    }
};

class Solution {
public:
    int myAtoi(string s) {
        Automaton automaon;
        for (auto c : s) {
            automaon.Add(c);
        }
        return automaon.GetResult();
    }
};
