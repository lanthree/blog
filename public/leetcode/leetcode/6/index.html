6. Z 字形变换

原题链接

读完题后，首先把题示的例子手算一下，然后就想着找规律。确实也有规律，但是代码写出来难懂。

class Solution {
public:
    string convert(string s, int numRows) {
        if (numRows == 1)
            return s;
        
        int delta = numRows*2-2;
        string ans;
        for (int i = 0; i < numRows; ++i) {
            for (int j = 0; j + i < s.size(); j += delta) {
                ans += s[j + i];
                
                if (i == 0 || i == numRows-1)
                    continue;
                
                if (j + delta - i < s.size())
                    ans += s[j + delta - i];
            }
        }
        return ans;
    }
};

看了题解后，其实有一个朴素的思路，就是按题目的描述，遍历字符串按Z字形拼接到各行，最后再把所有行拼接起来即可。

class Solution {
public:
    string convert(string s, int numRows) {
        if (numRows == 1) return s;

        vector<string> rows(min(int(s.size()), numRows));
        
        int cur_row = 0;
        bool is_down = false;
        for (int i = 0; i < s.size(); ++i) {
            rows[cur_row] += s[i];
            if (cur_row == 0 || cur_row == numRows-1)
                is_down = !is_down;
            
            cur_row += is_down? 1:-1;
        }

        string ans;
        for (auto row : rows) ans += row;
        return ans;
    }
};
