208. 实现 Trie (前缀树)

原题链接

class Trie {
private:
    struct TrieNode {
        bool is_word;
        vector<TrieNode*> children;

        TrieNode() : is_word(false), children(26, nullptr) {}
    };
    TrieNode* root;

public:
    /** Initialize your data structure here. */
    Trie() {
        root = new TrieNode();
    }
    
    /** Inserts a word into the trie. */
    void insert(string word) {
        int idx = 0;
        TrieNode* curnode = root;
        while (idx < word.length()) {
            int c_idx = word[idx] - 'a';
            if (curnode->children[c_idx] == nullptr) {
                curnode->children[c_idx] = new TrieNode();
            }
            curnode = curnode->children[c_idx];
            idx++;
        }
        curnode->is_word = true;

        // string str;
        // print_trie(root, str);
    }

    void print_trie(TrieNode* node, const string& str) {
        if (node == nullptr) {
            return;
        }
        if (node->is_word) {
            cout << "iter word:" << str << endl;
        }
        for (int i = 0; i < 26; i++) {
            char c = 'a'+i;
            print_trie(node->children[i], str+string(1,c));
        }
    }
    
    /** Returns if the word is in the trie. */
    bool search(string word) {
        int idx = 0;
        TrieNode* curnode = root;
        while (idx < word.length() && curnode) {
            int c_idx = word[idx] - 'a';
            curnode = curnode->children[c_idx];
            idx++;
        }
        return curnode && curnode->is_word;
    }
    
    /** Returns if there is any word in the trie that starts with the given prefix. */
    bool startsWith(string prefix) {
        int idx = 0;
        TrieNode* curnode = root;
        while (idx < prefix.length() && curnode) {
            int c_idx = prefix[idx] - 'a';
            curnode = curnode->children[c_idx];
            idx++;
        }
        return curnode;
    }
};

class Trie {
private:
    vector<Trie*> children;
    bool isEnd;

    Trie* searchPrefix(string prefix) {
        Trie* node = this;
        for (char ch : prefix) {
            ch -= 'a';
            if (node->children[ch] == nullptr) {
                return nullptr;
            }
            node = node->children[ch];
        }
        return node;
    }

public:
    Trie() : children(26), isEnd(false) {}

    void insert(string word) {
        Trie* node = this;
        for (char ch : word) {
            ch -= 'a';
            if (node->children[ch] == nullptr) {
                node->children[ch] = new Trie();
            }
            node = node->children[ch];
        }
        node->isEnd = true;
    }

    bool search(string word) {
        Trie* node = this->searchPrefix(word);
        return node != nullptr && node->isEnd;
    }

    bool startsWith(string prefix) {
        return this->searchPrefix(prefix) != nullptr;
    }
};
