33. 搜索旋转排序数组

原题描述

先找到最小的数字，然后截成两节做二分查找：

class Solution {
public:
    int search(vector<int>& nums, int target) {
        int len = nums.size();
        if (len == 1)
            return nums[0] == target ? 0 : -1;

        // 先找到最小值
        int s = 0, e = len-1;
        int smallest_idx = -1;
        while (s <= e) {
            int mid = (s+e)/2;
            if (isSmallest(nums, mid)) {
                smallest_idx = mid;
                break;
            }

            if (s == mid) {
                ++s;
            } else if (e == mid) {
                ++e;
            } else if (nums[s] < nums[mid] && nums[e] < nums[mid]) {    
                s = mid+1;
            } else {
                e = mid-1;
            }
        }

        int result = binarySearch(nums, 0, smallest_idx-1, target);
        if (result != -1) {
            return result;
        }
        result = binarySearch(nums, smallest_idx, len-1, target);
        if (result != -1) {
            return result;
        }
        return -1;
    }

private:
    bool isSmallest(vector<int>& nums, int idx) {
        if (idx == 0) {
            return nums[0] < nums[1];
        } else if (idx == nums.size()-1) {
            return nums[idx] < nums[idx-1];
        } else {
            return nums[idx] < nums[idx-1] && nums[idx] < nums[idx+1];
        }
    }

    int binarySearch(vector<int>& nums, int s, int e, int target) {
        while (s >= 0 && e < nums.size() && s <= e) {
            int mid = (s+e)/2;
            if (nums[mid] == target)
                return mid;
            else if (nums[mid] > target)
                e = mid-1;
            else
                s = mid+1;
        }
        return -1;
    }
};

看完题解发现还是太年轻了，不用找最小元素来分割数组，直接二分即可：

class Solution {
public:
    int search(vector<int>& nums, int target) {
        int len = nums.size();
        if (len == 0) return -1;
        if (len == 1) return nums[0] == target ? 0 : -1;

        int l = 0, r = len-1;
        while (l <= r) {
            int mid = (l+r)/2;
            if (nums[mid] == target) {
                return mid;

            } else if (nums[0] <= nums[mid]) { // 在左边
                if (nums[0] <= target && target < nums[mid]) {
                    // 判断是否在0～mid中间
                    r = mid-1;
                } else if (target > nums[mid] || target < nums[0]) { // if 条件可以省略
                    l = mid+1;
                } 
            } else if (nums[0] > nums[mid]) { // 在右边
                if (nums[mid] < target && target <= nums[len-1]) {
                    // 判断是否在mid~len中间
                    l = mid+1;
                } else if(target < nums[mid] || target > nums[len-1]) { // if 条件可以省略
                    r = mid-1;
                } 
            }
        }
        return -1;
    }
};
