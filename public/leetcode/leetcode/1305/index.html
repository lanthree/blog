1305. 两棵二叉搜索树中的所有元素

原题链接

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

class TreeIter {
public:
    TreeIter(TreeNode* _root) : root(_root), iter(nullptr) {
        iter = _Put(root);
    }
    
    bool Next(int *val) {
        if (iter == nullptr) {
            return false;
        }
        
        *val = iter->val;
        if (iter->right) {
            iter = _Put(iter->right);
        } else if (iter_stack.empty()) {
            iter = nullptr;
        } else {
            iter = iter_stack.top();
            iter_stack.pop();
        }
        return true;
    }
    
private:
    TreeNode* _Put(TreeNode* root) {
        TreeNode* ret = root;
        while (ret && ret->left) {
            iter_stack.push(ret);
            ret = ret->left;
        }
        return ret;
    }
    
    
private:
    TreeNode* root;
    TreeNode* iter;
    stack<TreeNode*> iter_stack;
};

class Solution {
public:
    vector<int> getAllElements(TreeNode* root1, TreeNode* root2) {
        TreeIter iter1(root1), iter2(root2);
        int val1, val2;
        bool has1 = false, has2 = false;
        vector<int> ret;
        while (true) {
            if (!has1) {
                has1 = iter1.Next(&val1);
            }
            if (!has2) {
                has2 = iter2.Next(&val2);
            }
            
            if (!has1 && !has2) {
                break;
            } else if (has1 && has2) {
                if (val1 < val2) {
                    has1 = false;
                    ret.push_back(val1);                    
                } else {
                    has2 = false;
                    ret.push_back(val2);
                }
            } else if (has1) {
                has1 = false;
                ret.push_back(val1);
            } else {
                has2 = false;
                ret.push_back(val2);
            }
        }
        return ret;
    }
};
