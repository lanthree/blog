88. 合并两个有序数组

原题链接

一般思路可以通过，但时耗只击败60% 而且nums1原大小就是m+n这个条件没用到，感觉会有更优的算法

class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        vector<int> ans(m+n, 0);
        
        int ans_idx = 0;
        int n1_idx = 0, n2_idx = 0;
        while (n1_idx < m || n2_idx < n) {
            int n1 = n1_idx < m ? nums1[n1_idx] : INT_MAX;
            int n2 = n2_idx < n ? nums2[n2_idx] : INT_MAX;
            if (n1 < n2) {
                ans[ans_idx++] = n1;
                n1_idx++;
            } else {
                ans[ans_idx++] = n2;
                n2_idx++;
            }
        }
        nums1 = ans;
    }
};

再想了一下，只要从后往前填num1，就不会有覆盖原数组的问题，也不用再建一个中加数组了:

class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {      
        int ans_idx = m+n-1;
        int n1_idx = m-1, n2_idx = n-1;
        while (n1_idx >= 0 || n2_idx >= 0) {
            int n1 = n1_idx >= 0 ? nums1[n1_idx] : INT_MIN;
            int n2 = n2_idx >= 0 ? nums2[n2_idx] : INT_MIN;
            if (n1 > n2) {
                nums1[ans_idx--] = n1;
                n1_idx--;
            } else {
                nums1[ans_idx--] = n2;
                n2_idx--;
            }
        }
    }
};

其实可以再优化一点，因为本来就是往nums1填充，可以把循环寻条件中的n1_idx >= 0去掉:

class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {      
        int ans_idx = m+n-1;
        int n1_idx = m-1, n2_idx = n-1;
        while (n2_idx >= 0) {
            int n1 = n1_idx >= 0 ? nums1[n1_idx] : INT_MIN;
            int n2 = nums2[n2_idx];
            if (n1 > n2) {
                nums1[ans_idx--] = n1;
                n1_idx--;
            } else {
                nums1[ans_idx--] = n2;
                n2_idx--;
            }
        }
    }
};

击败100%就出现了
