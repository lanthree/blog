69. x 的平方根

原题链接

看上去是数学题，但是并不知道数学上求平方根的方法。可以使用二分法进行搜索：

class Solution {
public:
    int mySqrt(int x) {
        if (x == 0 || x == 1)
            return x;

        int l = 1, r = x/2;
        int ans = 1;
        while (l <= r) {
            long long mid = l+(r-l)/2;
            long long mul = mid*mid;
            if (mul == x) {
                return mid;
            } else if (mul > x) {
                r = mid-1;
            } else {
                ans = mid;
                l = mid+1;
            }
        }
        return ans != 1 ? ans : r;
    }
};

官方题解果然有数学方法：牛顿迭代。具体方法还是看题解吧~
