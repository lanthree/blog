204. 计数质数

原题链接

素数筛，注意题目中“统计所有小于非负整数 n 的质数的数量”的小于。

class Solution {
public:
    int countPrimes(int n) {
        int max_v = n-1;
        if (max_v < 2) return 0;

        vector<bool> is_prime(max_v, true);
        int ret = max_v-1;
        for (int i = 2; i < max_v; i++) {
            if (not is_prime[i]) {
                continue;
            }
            int bc = 2;
            int bs = i*bc;
            for (int bs =i*bc; bs <= max_v; bs = i*(++bc)) {
                if (not is_prime[bs]) {
                    continue;
                }
                is_prime[bs] = false;
                ret -= 1;
            }
        }
        return ret;
    }
};

我这里是用减法统计，题解用加法统计：

class Solution {
public:
    int countPrimes(int n) {
        vector<int> isPrime(n, 1);
        int ans = 0;
        for (int i = 2; i < n; ++i) {
            if (isPrime[i]) {
                ans += 1;
                if ((long long)i * i < n) {
                    for (int j = i * i; j < n; j += i) {
                        isPrime[j] = 0;
                    }
                }
            }
        }
        return ans;
    }
};
